curl -X POST http://localhost:8000/api/ride/request \
     -H "Authorization: Bearer YOUR_PASSENGER_TOKEN" \
     -H "Content-Type: application/json" \
     -d '{
           "pickup_location": "Central Park",
           "drop_location": "Airport",
           "initial_fare": 500
         }'
